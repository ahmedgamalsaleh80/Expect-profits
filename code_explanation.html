def normalise(X):
    means, stds = X.mean(0), X.std(0)
    stds[stds == 0] = 1
    return (X - means) / stds
Z = (X − μ) / σ     μ = المتوسط     σ = الانحراف المعياري

# score = w·x + b ؛ لو y*score < 1 → حدّث بالكامل، غير كده → regularisation فقط
def svm_train(X, y, lr=0.01, reg=0.01, iters=300):
    w, b = np.zeros(X.shape[1]), 0.0
    for _ in range(iters):
        for xi, yi in zip(X, y):
            if yi * (np.dot(w, xi) + b) < 1:
                w += lr * (yi * xi - 2*reg*w)
                b += lr * yi
            else:
                w -= lr * 2*reg*w
    return w, b
🔥
ده قلب المشروع! الـ SVM بيتعلم هنا. اللي بيحصل: الموديل بياخد كل نقطة، بيحسب إذا كانت مصنفة صح، لو لأ بيصحح نفسه.

📝 شرح كل parameter:

lr=0.01
Learning Rate
معدل التعلم — بيحدد خطوة التصحيح كام كبيرة. كبير جداً = هيتخطى الحل. صغير جداً = هياخد وقت طويل. 0.01 قيمة ثابتة محترمة.
reg=0.01
Regularization (λ)
معامل التنظيم — بيمنع الـ Overfitting. بيعاقب الـ weights الكبيرة جداً عشان الموديل متحفظش الداتا بس يتعلم منها.
iters=300
Iterations
عدد مرات التدريب — الموديل هيلف على كل الداتا 300 مرة. كل مرة بيتعلم أكتر. 300 قيمة معقولة.

سطر 10

إذا y·(w·x + b) < 1:   w = w + α·(y·x − 2λ·w)     b = b + α·y
إذا y·(w·x + b) ≥ 1:   w = w − α·2λ·w         (Regularization فقط)

def k_means(X, k, iters=100):
    centroids = X[random.sample(range(len(X)), k)].copy()
    labels = np.zeros(len(X), dtype=int)
    for _ in range(iters):
        labels = np.argmin(np.sum((X[:, None] - centroids)**2, axis=2), axis=1)
        for ci in range(k):
            if (labels == ci).any():
                centroids[ci] = X[labels == ci].mean(0)
    return labels, centroids

def evaluate(preds, y):
    tp = ((preds==1) & (y==1)).sum()
    fp = ((preds==1) & (y==-1)).sum()
    fn = ((preds==-1) & (y==1)).sum()
    acc  = (preds == y).mean()
    prec = tp/(tp+fp) if tp+fp else 0
    rec  = tp/(tp+fn) if tp+fn else 0
    f1   = 2*prec*rec/(prec+rec) if prec+rec else 0
    return acc, prec, rec, f1

📝 شرح المقاييس الأربعة:

TP
preds==1 AND y==1
True Positive: الموديل قال Profit والحقيقة Profit ✅ — صح في الإيجابي.
FP
preds==1 AND y==-1
False Positive: الموديل قال Profit بس الحقيقة Loss ❌ — إنذار كاذب.
FN
preds==-1 AND y==1
False Negative: الموديل قال Loss بس الحقيقة Profit ❌ — فاتته حاجة.
Accuracy
(preds==y).mean()
نسبة الصح من الكل. 0.85 = الموديل صح في 85% من الحالات.
Precision
TP / (TP + FP)
من اللي قلنا عليهم Profit، كام منهم Profit فعلاً؟
Recall
TP / (TP + FN)
من كل الـ Profits الحقيقية، كام منهم اكتشفنا؟
F1
2·P·R / (P + R)
المقياس الأشمل: المتوسط التوافقي بين Precision وRecall. أحسن من Accuracy لو الداتا مش متوازنة.

def main():
    path = sys.argv[1] if len(sys.argv) > 1 else "data.csv"
    try:
        df = load_csv(path)
    except FileNotFoundError:
        print(f"⚠ مش لاقي الملف: {path}"); return

    cols = detect_numeric_cols(df)
    features, target = cols[:-1], cols[-1]

    X, y, sub = prepare_data(df, features, target)
    X_norm     = normalise(X)

    # Labels: فوق المتوسط = Profit (+1)، تحت = Loss (-1)
    tmean = y.mean()
    y_svm = np.where(y >= tmean, 1, -1)

    # تقسيم 80/20
    idx = list(range(len(X_norm))); random.shuffle(idx)
    n_tr = int(len(idx)*0.8)
    tr, te = idx[:n_tr], idx[n_tr:]

    w, b = svm_train(X_norm[tr], y_svm[tr])
    preds = svm_predict(w, b, X_norm[te])
    acc, prec, rec, f1 = evaluate(preds, y_svm[te])

    km_labels, _ = k_means(X_norm, k=3)

    plot_results(...)

if __name__ == "__main__":
    main()

with st.spinner("⚙ جاري التحليل..."):

    X, y, sub = prepare_data(df, features, target)
    X_norm    = normalise(X)
    tmean     = y.mean()
    y_svm     = np.where(y >= tmean, 1, -1)

    idx = list(range(len(X_norm))); random.shuffle(idx)
    n_tr = int(len(idx)*split)       # split من الـ slider
    tr, te = idx[:n_tr], idx[n_tr:]

    w, b  = svm_train(X_norm[tr], y_svm[tr])
    preds = svm_predict(w, b, X_norm[te])
    acc, prec, rec, f1 = evaluate(preds, y_svm[te])

    km_labels, _ = k_means(X_norm, k)

svm_predict تصنيف نقاط جديدة w, b, X array من +1 و −1
k_means تجميع K-Means من الصفر X, k, iters labels, centroids
evaluate حساب Accuracy/Precision/Recall/F1 preds, y acc, prec, rec, f1
plot_results رسم 4 مخططات Dashboard كل النتائج صورة PNG / عرض في Streamlit
